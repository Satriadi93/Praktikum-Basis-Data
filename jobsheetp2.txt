nyalakan xampp dan start mySQL dan Apache
Buka CMD dan masuk ke root dengan cara ketik
cd .. 3x
cd xampp/mysql/bin
mysql -u root
---------------------------------------------------------------------------
Untuk membuat data base 
create database nama_database;
---------------------------------------------------------------------------
Untuk melihat database 
show databases;
---------------------------------------------------------------------------
Masuk atau menggunakan database
use nama_database
---------------------------------------------------------------------------
Membuat table 
create table supplier(nama_kolom1 type_data, nama_kolom2 type_data, nama_kolom3 type_data, primary key (nama_kolom));
---------------------------------------------------------------------------
Membuat index
create index `nama_index` on `nama_kolom`(`nama_baris`);
---------------------------------------------------------------------------
0. Membuat stored procedure menampilkan isi kolom;
delimiter //
create procedure nama_procedure()
begin
select*from nama_kolom;
end //
delimiter ;
call nama_procedure();

0. membuat stored procedure menampilkan isi kolom sesuai inputan;
delimiter //
create procedure nama_procedure( IN nama_inputan type_data)
begin
select*from nama_kolom where nama_baris=nama_inputan;
end //
delimiter ;
call nama_procedure(inputan);

0. membuat stored procedure mengisi isi dari tabel;
delimiter //
create procedure nama_procedure( IN nama_inputan1 type_data, IN nama_inputan2 type_data)
begin
insert into nama_tabel values(nama_kolom1, nama_kolom2);
end //
delimiter ;
call nama_procedure(input1, input2);
